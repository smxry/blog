	name: "gshopHeader",

        default:'默认值'

# 3.meta传参

公共组件在某一个组件中不需要，则通过route.meta来判断

router/index.js

{
      path: '/home',//需要公共组件
      component: () => import('../pages/home/home'),
      meta:{//通过meta传参
        isFooterContainer:true
      }
    },
    {
      path: '/login',//不需要公共组件
